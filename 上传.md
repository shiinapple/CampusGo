rm -f /root/campus/target/CampusGo-app.jar

删掉docker
cd /root/campus/dev-ops
docker-compose -f docker-compose-app.yml down -v --remove-orphans
docker rm -f CampusGo
docker rmi -f system/campusgo-app:1.0-snapshot
rm -f /root/campus/target/CampusGo-app.jar
docker builder prune -a -f

运行docker镜像
cd /root/campus/dev-ops
docker-compose -f docker-compose-environment-aliyun.yml up -d
